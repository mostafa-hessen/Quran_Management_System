Clean Code & Reusability

⚡ القاعدة الذهبية: فصل المسؤوليات — الـ Page بتعرض فقط، الـ Hook بيجيب البيانات، الـ Component بيعرض فقط.
مثال عملي — useTeachers Hook
❌ الطريقة الغلط 
// TeachersPage.tsx — كل حاجة في مكان واحد
export default function TeachersPage() {
  const [teachers, setTeachers] = useState([])
  const [loading, setLoading]   = useState(false)
  const [error, setError]       = useState(null)

  useEffect(() => {
    setLoading(true)
    supabase.from("teachers").select("*")
      .then(({ data }) => setTeachers(data))
      .finally(() => setLoading(false))
  }, [])

  // 200 سطر هنا ...
}
✅ الطريقة الصح 
// hooks/useTeachers.ts — المسؤولية: جلب البيانات فقط
export function useTeachers() {
  return useQuery({
    queryKey: ["teachers"],
    queryFn: async () => {
      const { data, error } = await supabase
        .from("teachers")
        .select("*, teacher_phones(*)")
      if (error) throw error
      return data
    }
  })
}

export function useCreateTeacher() {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (data) => supabase.from("teachers").insert(data),
    onSuccess: () => queryClient.invalidateQueries(["teachers"])
  })
}

// pages/TeachersPage.tsx — المسؤولية: عرض فقط ✅
export default function TeachersPage() {
  return (
    <div>
      <PageHeader title="المعلمون" />
      <TeacherList />
    </div>
  )
}

Reusable DataTable Component
// component واحد بيتستخدم في كل الصفحات
interface DataTableProps<T> {
  data:      T[]
  columns:   ColumnDef<T>[]
  loading?:  boolean
  onRowClick?: (row: T) => void
}

export function DataTable<T>({ data, columns, loading }: DataTableProps<T>) {
  if (loading) return <LoadingSpinner />
  return <table>...</table>
}

// الاستخدام — نفس الـ component:
<DataTable data={teachers} columns={teacherColumns} loading={isLoading} />
<DataTable data={students} columns={studentColumns} loading={isLoading} />
<DataTable data={payments} columns={paymentColumns} loading={isLoading} />
